[Unit]
Description=Camera Streaming Server
Documentation=https://github.com/markatango/stream-android-camera-to-website
After=network.target
Wants=network.target

[Service]
Type=simple
User=www-data
Group=www-data
WorkingDirectory=/opt/camera-app/camera-streaming-server
ExecStart=/usr/bin/node server.js
Restart=always
RestartSec=10
StandardOutput=journal
StandardError=journal
SyslogIdentifier=camera-streaming-server

# Environment variables
Environment=NODE_ENV=production
Environment=PORT=3001
EnvironmentFile=-/opt/camera-app/camera-streaming-server/.env

# Security settings
NoNewPrivileges=true
PrivateTmp=true
ProtectSystem=strict
ProtectHome=true
ReadWritePaths=/opt/camera-app/camera-streaming-server
CapabilityBoundingSet=CAP_NET_BIND_SERVICE

# Resource limits
LimitNOFILE=65536
LimitNPROC=4096

# Process management
KillMode=mixed
KillSignal=SIGTERM
TimeoutStopSec=30

[Install]
WantedBy=multi-user.target
